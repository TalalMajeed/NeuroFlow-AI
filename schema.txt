CREATE TABLE "users" (
    "userID" VARCHAR(5) PRIMARY KEY,
    "name" VARCHAR(45),
    "gender" VARCHAR(6),
    "email" VARCHAR(45),
    "password" VARCHAR(45),
    "description" VARCHAR(100),
    "occupation" VARCHAR(45),
    "image" BYTEA
);

CREATE TABLE "diagrams" (
    "diagramID" VARCHAR(5) PRIMARY KEY,
    "userID" VARCHAR(5),
    "name" VARCHAR(45),
    "description" VARCHAR(100),
    "data" BYTEA,
    "date" DATE,
    FOREIGN KEY ("userID") REFERENCES "users"("userID")
);
